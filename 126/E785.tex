On considère la série entière $S(x)=\sum_{n\in\N} \frac{a_n}{n!} x^n$.
\begin{enumerate}
  \item Montrer que le rayon de convergence de $S$ est $+\infty$.
  \item Montrer que
    \begin{equation*}
      \int_0^{+\infty} S(x) e^{-x}\d x = \sum_{n\in\N} a_n.
    \end{equation*}
\end{enumerate}
