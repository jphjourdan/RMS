\begin{enumerate}
  \item Montrer que la suite $\left( P\left( X\geq n \right)  \right)$ satisfait une relation de récurrence d'ordre $2$.
    En déduire la loi de $X$.

  \item Montrer que $X$ est d'espérance finie et calculer $E(X)$.
\end{enumerate}
