\begin{exercise}[ID={RMS134 E668},subtitle={Mines-Ponts MP 2023},tags={analyse, mines-ponts, 2023},difficulty={}]
  Soient $(E,\|\;\|)$ un espace vectoriel normé et $A$ une partie non
 vide de $E$.

 Soit $f:x\in E\mapsto d(x,A)=\inf\{\|x-a\|, a\in A\}$.
 \begin{enumerate}[\bfseries a)]
 \item Montrer que $f$ est $1$-lipschitzienne.
 \item Montrer que $A$ est fermé si et seulement si
   $A=f^{-1}(\{0\})$.
 \item Montrer que tout fermé de $E$ est intersection décroissante
   d'ouverts.
 \item Montrer que tout ouvert est union croissante de fermés.
 \end{enumerate}
  % exo 668 RMS 134-1
\end{exercise}
\begin{solution}
  \begin{enumerate}[\bfseries a)]
  \item ok
  \item On peut montrer en fait que $f^{-1}(\{0\})=\overline{A}$.
  \item On considère $(f^{-1}(]-\infty,1/n[)$.
  \end{enumerate}
\end{solution}
\end{enonce}
