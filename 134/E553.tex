\begin{exercise}[ID={RMS134 E553},subtitle={Oral Mines-Ponts},difficulty={},tags={algebre, mines-ponts, 2023}]
  Soient $E$ un espace vectoriel de dimension finie et $u\in\Lin(E)$.
\begin{enumerate}[\bfseries a)]
\item Montrer l'équivalence des trois propriétés suivantes:
  \[(i)\;\Im(u)=\Im(u^2)\qquad(ii)\;\Ker(u)=\Ker(u^2)\qquad;(iii)E=\Im(u)\oplus\Ker(u)\]
\item Donner des exemples d'endomorphismes vérifiant ces propriétés.
\item L'équivalence est-elle vraie en dimension infinie? Montrer que
  $(i)$ et $(ii)$ équivaut à $(iii)$
\end{enumerate}
   %exo 553 RMS 134-1
\end{exercise}
\begin{solution}
  \begin{enumerate}[\bfseries a)]
  \item Classique.
  \item Un projecteur par exemple.
  \item Sans hypothèse de dimension, on a $(i)\eq E=\Im(u)+\Ker(u)$ et
    $(ii)\eq \Im(u)\cap\Ker(u)=\{0\}$, mais pas $(i)\eq (ii)$. Par
    exemple pour $u:P\mapsto P'$ sur $\K[X]$, on a $\Im(u^n)=\K[X]$
    mais $\Ker(u^n)=\K_n[X]$.
  \end{enumerate}
\end{solution}
\end{enonce}
