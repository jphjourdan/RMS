\begin{exercise}[ID={RMS134 E1525},subtitle={IMT MP 2023},theme={analyse},concours={mines-telecom},annee={2023},filiere={MP}, difficulty={0}]
  Soit $(1)$ l'équation différentielle $xy'+y=\e^x$.
 \begin{enumerate}
 \item Trouver les solutions de $(1)$ développables en série entière
   au voisinage de $0$.
 \item Les solutions de $(1)$ sur $]0,+\infty[$ sont-elles toutes
   développables en série entière au voisinage de $0$ ?
 \item Résoudre $(1)$ sur un intervalle $I$ de $\R$. Discuter suivant
   $I$.
 \item On ajoute à l'équation $(1)$ la condition $y(x_0)=y_0$ (avec
   $x_0,y_0\in\R$) pour obtenir un problème numéroté $(2)$. Que dit le
   théorème de Cauchy à propos du problème $(2)$ si on travaille sur
   $]0,+\infty[$ ? Résoudre $(2)$.
 \item Représenter graphiquement la ou les solutions développables en
   série entière.
 \end{enumerate}
