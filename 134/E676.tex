\begin{exercise}[ID={RMS134 E676},subtitle={Mines-Ponts MP 2023},tags={analyse, mines-ponts, 2023},difficulty={}]
  Soient $(E,\|\;\|)$ un espace normé réel, $p\in\N^*$,
 $(x_1,\ldots,x_p)\in E^p$.
 \begin{enumerate}[\bfseries a)]
 \item Montrer que $(x_1,\ldots,x_p)$ est libre si et seulement si
   \[\inf\left\{\left\|\sum_{i=1}^p\lambda_ix_i\right\|\;;\;
       (\lambda_1,\ldots,\lambda_p)\in\R^p\right\}>0\]
 \item En déduire que l'ensemble des $(x_1,\ldots,x_p)\in E^p$ tels
   que $(x_1,\ldots,x_p)$ est libre est un ouvert de $E^p$. Retrouver
   ce résultat plus simplement si $E$ est de dimenstion finie.
 \end{enumerate}

\end{exercise}
