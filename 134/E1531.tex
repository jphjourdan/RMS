\begin{exercise}[ID={RMS134 E1531},subtitle={CCINP MP 2023},difficulty={},tags={probas, ccinp, 2023}]
  Une personne sur une échelle est en train de peindre un bâtiment. La
 probabilité qu'un passant reçoive une goutte de peinture est
 $p\in]0,1[$. On note $X$ (resp. $Y$) le nombre de passants ayant reçu
 une goutte de peinture (resp. n'ayant pas reçu de goutte.)
 \begin{enumerate}[\bfseries a)]
 \item On suppose que $n$ personnes sont passées. Donner les lois de
   $X$ et de $Y$. Sont-elles indépendantes ?
 \item On note à présent $N$ le nombre de passants dans la journée. On
   suppose que $N$ suit une loi de Poisson de paramètre
   $\lambda$. Donner la loi de $X$ et de $Y$. Donner l'espérance et la
   variance de $X$.
 \item Montrer que $X$ et $Y$ sont indépendantes. Calculer
   $\Cov(X,Y)$.
 \end{enumerate}

\end{exercise}
