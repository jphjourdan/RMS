  Soient $x\in\left]0,4\right[$ et $\omega = \arcsin\left( \frac{x}{2} - 1 \right) + \frac{\pi}{2}$.
  \begin{enumerate}
    \item Montrer que $\cos(\omega) = 1 - \frac{x}{2}$.

    \item Montrer que $\cos\left( \frac{\omega}{2} \right)$ et $\sin\left( \frac{\omega}{2} \right)$ sont positifs.

    \item Montrer que
      \begin{equation*}
        \cos\left( \frac{\omega}{2} \right) = \frac{\sqrt{4-x}}{2}
        \quad\et\quad
        \sin\left( \frac{\omega}{2} \right) = \frac{\sqrt{x}}{2}.
      \end{equation*}


    \item Montrer que
      \begin{equation*}
        \arcsin\left( \frac{x}{2} - 1 \right) + \frac{\pi}{2}
        =
        2\arctan\left( \sqrt{\frac{x}{4-x}} \right).
      \end{equation*}
  \end{enumerate}

\end{exercise}
