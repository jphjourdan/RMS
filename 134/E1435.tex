\begin{exercise}[ID={RMS134 E1435},subtitle={IMT MP 2023},difficulty={},tags={algebre, mines-telecom, 2023}]
  On pose $E=\mathcal{C}^1([0,1],\R)$. Pour $(f,g)\in E^2$, on pose
  $\langle f,g\rangle=\dint_0^1(fg+f'g')$.
  \begin{enumerate}[\bfseries a)]
  \item Montrer que $\langle\;,\;\rangle$ est un produit scalaire.

    Soient $V=\{f\in E, f(0)=f(1)=0\}$ et $W=\{f\in E, f''=f\}$.
  \item Montrer que $V$ et $W$ sont des sous-espaces vectoriels puis
    que $\{t\mapsto\e^t, t\mapsto\e^{-t}\}$ est une base de $W$.
  \item Montrer que $V$ et $W$ sont orthogonaux.
  \item Calculer $p_W(f)$ le projeté orthogonal de $f\in E$ sur $W$.
  \item Montrer que $V$ et $W$ sont supplémentaires.
  \end{enumerate}

\end{exercise}
