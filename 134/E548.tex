\begin{exercise}[ID={RMS134 E548},subtitle={Oral Mines-Ponts},theme={algebre},annee={2023},concours={mines-ponts},filiere={MP}]
  Soient $x,y,z$ des rationnels non nuls. Montrer que la matrice
  $
  \begin{pmatrix}
    x & y & z\\
    2y & z & 2x \\
    z & x & 2y
  \end{pmatrix}$ est inversible.
   %exo 548 RMS 134-1
\end{exercise}
\begin{solution}
  Montrer que les deux premières colonnes sont libres puis que la
  troisième n'est pas une combinaison linéaire des deux premières ?
\end{solution}
\end{enonce}
