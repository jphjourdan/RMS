 \begin{enumerate}
 \item Justifier que, pour tout $M\in\M_n(\C)$, $\Sp(M)=\Sp(M^T)$.
 \item Montrer que $A$ est inversible si et seulement si
   $1\notin\Sp(A)$.
 \item Montrer que le polynôme $X^4-2X^2+X$ est annulateur de $A$. La
   matrice $A$ est-elle diagonalisable?
 \end{enumerate}
