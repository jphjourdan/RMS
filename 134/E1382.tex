\begin{exercise}[ID={RMS134 E1382},subtitle={CCINP MP
    2023},theme={algebre},concours={ccinp},annee={2023},filiere={MP}, difficulty={0}]
  \begin{enumerate}
  \item Soient $a,b,c\in\Z^*$ avec $a\wedge b=1$. Montrer que, si $a$
    divise $c$ et $b$ divise $c$, alors $ab$ divise $c$.
  \item Trouver une solution de
    $(*) : x\equiv 6\,[17]\text{ et }x\equiv 4\,[15]$.
  \item Trouver toutes les solutions de $(*)$.
  \end{enumerate}
  %exo 1382 RMS 134-1
\end{exercise}
\begin{solution}
  \begin{enumerate}
  \item Identité de Bézout.
  \item On trouve $244$ ou bien $-11$ si on pense à descendre.
  \item $x$ solution ssi $x\equiv -11\,[17]$ et $x\equiv -11\,[15]$,
    donc ssi $17\times 15=255$ divise $x+11$.
    
    On a donc $\mathcal{S}=\{255k-11, k\in\Z\}$.
  \end{enumerate}
\end{solution}
\end{enonce}
