  Soit $(X,Y)$ un couple de variables aléatoires à valeurs dans $\N^2$
  tel qu'il existe $\alpha\in\R$ vérifiant,
  $\forall (k,\ell)\in\N^2,
  \P(X=k,Y=\ell)=\dfrac{\alpha}{2^{k+\ell}}$.
  \begin{enumerate}
  \item Trouver $\alpha$. Les variables $X$ et $Y$ sont-elles
    indépendantes ?
  \item Calculer $G_X(t)$, $\Esp(X)$, $\Var(X)$ et $\Cov(X,Y)$.
  \item Calculer $\P(X\ge k)$ pour tout $k\in\N$ et retrouver
    $\Esp(X)$.
  \item On pose $Z=\min(X,Y)$. Déterminer la loi de $Z$.
  \item Calculer $\P(X\ge Y)$.
  \end{enumerate}

\end{exercise}
