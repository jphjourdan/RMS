  Soit $A=\begin{pmatrix} 3&-1&2\\ 2&0&2\\ 1&-1&2 \end{pmatrix}$.
    \begin{enumerate}
      \item Montrer que $A$ a une valeur propre double $a>0$ et une simple $b>0$.
        La matrice $A$ est-elle diagonalisable?

      \item Soit $f$ une fonction de $\R>$ dans $\R$ de classe $\class^2$.
        Montrer qu'il existe un unique polynôme $P_f\in\R_2[X]$ tel que
        \begin{align*}
          P_f(a)&= f(a),&
          P_f(b)&= f(b),&
          P_f'(a)&= f'(a).
        \end{align*}

      \item Pour toute fonction $f\in\class^2(\R>,\R)$, on pose $f(A) = P_f(A)$.
        Calculer $f(A)$ dans les cas où $f:x\mapsto x^2$ et $f:x\mapsto x^3$.

      \item Désormais, on prend $f:x\mapsto \frac{1}{x}$.
        Conjecturer la valeur de $A f(A)$ et prouver cette conjecture.
    \end{enumerate}
\end{exercise}
