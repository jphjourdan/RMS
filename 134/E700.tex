\begin{exercise}[ID={RMS134 E700},subtitle={Mines-Ponts MP 2023},tags={analyse, mines-ponts, 2023},difficulty={}]
  Soit $(a_n)_{n\ge 0}$ une suite réelle décroissante de réels
 strictement positifs, telle que $a_0=1$. On pose
 $b_n=\dsum_{k=1}^n\left(1-\dfrac{a_{k-1}}{a_k}\right)\dfrac1{a_k}$
 pour tout $n\ge 1$.
 \begin{enumerate}[\bfseries a)]
 \item Montrer que $b_n\in [0,1]$ pour tout $n\ge 1$.
 \item On fixe $\ell\in [0,1]$. Montrer que l'on peut choisir la suite
   $(a_n)_{n\ge 0}$ de telle sorte que $b_n\to\ell$.
 \end{enumerate}

\end{exercise}
