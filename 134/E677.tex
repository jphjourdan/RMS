\begin{exercise}[ID={RMS134 E677},subtitle={Mines-Ponts MP 2023},theme={analyse},annee={2023},concours={mines-ponts},filiere={MP}]
  Soient $n\ge 2$, $K$ un compact de $\R^n$ et $\varepsilon>0$. Une
 partie $A\subset K$ est $\varepsilon$-séparée si, pour tous $x,y\in
 A$ tel que $\|x-y\|<\varepsilon$, on a $x=y$.
 \begin{enumerate}
 \item Montrer qu'il existe un entier $M(\varepsilon)$ tel que toute
   partie $\varepsilon$-séparée de $K$ est de cardinal inférieur à
   $M(\varepsilon)$ et il existe une partie $\varepsilon$-séparée de
   $K$ de cardinal $M(\varepsilon)$.
 \item Soit $f:K\to\K$. On suppose que, pour tous $x,y\in K$,
   $\|f(x)-f(y)\|=\|x-y\|$. Montrer que $f$ est surjective.
 \end{enumerate}

\end{exercise}
