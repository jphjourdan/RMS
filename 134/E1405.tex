\begin{exercise}[ID={RMS134 E1405},subtitle={CCINP MP 2023},difficulty={},tags={algebre, ccinp, 2023}]
  Soit $u\in\Lin(\R^3)$ non nul tel que $u^3+u=0$.
  \begin{enumerate}[\bfseries a)]
  \item Montrer que $E=\Ker(u)\oplus\Im(u)$ et que
    $\Im(u)=\Ker(u^2+id)$.
  \item Montrer que $u$ n'est pas injective, puis que $\rg(u)=2$.
  \item Montrer qu'il existe une base de $E$ dans laquelle la matrice
    de $u$ est $
    \begin{pmatrix}
      0 & 0 & 0\\
      0 & 0 & -1\\
      0 & 1 & 0
    \end{pmatrix}$.
  \end{enumerate}
