\begin{exercise}[ID={RMS134 E1394},subtitle={CCINP MP 2023},theme={algebre},concours={ccinp},annee={2023},filiere={MP}]
  Soit $f:\C\to\C$.
 \begin{enumerate}
 \item Montrer que $f$ est un endomorphisme de $\C$ considéré comme un
   $\R$-espace vectoriel si et seulement s'il existe deux nombres
   complexes $a,b$ tels que $\forall z\in\C,\;f(z)=az+b\conj{z}$.
 \item Montrer l'unicité du couple $(a,b)$.
 \item Montrer que $f$ est un projecteur si et seulement si
   $a^2+|b|^2=a$ et $b(a+\conj{a})=b$.
 \item Montrer que $f$ est un projecteur différent de l'endomorphisme
   nul et de l'identité si et seulement si $\Re(a)=\dfrac12$ et
   $|a|=|b|$.
 \end{enumerate}
