\begin{exercise}[ID={RMS134 E1396},subtitle={IMT MP 2023},difficulty={},tags={algebre, mines-telecom, 2023}]
  Soit $f\in\Lin(\R^4)$ telle que $f\circ f=0$. Montrer que $\rg(f)\le 2$.
  %exo 1396 RMS 134-1
\end{exercise}
\begin{solution}
  On a $\Im(f)\subset\Ker(f)$ et donc $\rg(f)\le \dim(\Ker(f))$. Le
  théorème du rang donne alors $4=\rg(f)+\dim(\Ker(f))\ge 2\rg(f)$,
  d'où le résultat.
\end{solution}
\end{enonce}
