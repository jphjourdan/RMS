\begin{exercise}[ID={RMS134 E1430},subtitle={CCINP MP 2023},theme={algebre},concours={ccinp},annee={2023},filiere={MP}, difficulty={0}]
  Soient $a\in\R^*$, $U=(a^{j-i})_{1\le i,j\le n}\in\M_n(\R)$ et
 $u\in\Lin(\R^n)$ canoniquement associé à $U$.
 \begin{enumerate}
 \item Déterminer le rang de $u$ et son déterminant.
 \item Déterminer la dimension du noyau de $u$ ainsi qu'une équation
   de ce noyau.
 \item Déterminer la dimension de l'image de $u$ et une base de cette
   image.
 \item Étudier la diagonalisabilité de $u$.
 \item Pour $k\in\N^*$, exprimer $U^k$ en fonction de $U$.
 \item Déterminer le polynôme minimal de $u$ et retrouver le résultat
   de la question précédente.
 \end{enumerate}
