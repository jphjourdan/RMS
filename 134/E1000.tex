\begin{exercise}[ID={RMS134 E1000},subtitle={Mines-Ponts PC 2023},tags={colle},difficulty={}]
  Soit $(a,x,y,z)\in\R^4$.
  Calculer le déterminant de 
  \begin{equation*}
    \begin{pmatrix}
      a & x & \cdots & \cdots & x\\
      y & z & 0 & \cdots & 0\\
      \vdots & 0 &\ddots &\ddots & \vdots\\
      \vdots & \vdots &\ddots &\ddots &0\\
      y & 0 & \cdots & 0 & z
    \end{pmatrix} 
    \in\mat_n(\R)
  \end{equation*}
  pour $n\geq 2$.
\end{exercise}
\begin{solution}
  Il semble que la relation est $D_n = z D_{n-1} + z^{n-2}yx$, ce qui s'écrit également
  \begin{equation*}
    \frac{D_n}{z^n} = \frac{D_{n-1}}{z_{n-1}} + \frac{yx}{z^2}.
  \end{equation*}
\end{solution}
\end{enonce}
