  Soient $\varphi\in\R$ et $M_n=(m_{i,j})_{1\le i,j\le n}\in\M_n(\R)$
 définie par $m_{i,j}=2\cos(\varphi)$ si $i=j$, $m_{i,j}=1$ si
 $|i-j|=1$, les autres coefficients étant nuls.
 \begin{enumerate}[\bfseries a)]
 \item On suppose que $\varphi\notin\pi\Z$. Trouver une relation de
   récurrence vérifiée par $D_n=\det(M_n)$ et exprimer $D_n$.
 \item Déterminer $D_n$ lorsque $\varphi\in\pi\Z$.
 \end{enumerate}
