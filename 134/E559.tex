\begin{exercise}[ID={RMS134 E559},subtitle={Oral Mines-Ponts},difficulty={},tags={algebre, mines-ponts, 2023}]
  \begin{enumerate}[\bfseries a)]
  \item Déterminer le rang de $\mathrm{Com}(A)$ en fonction du rang de
    $A$.
  \item Calculer $\mathrm{Com}(\mathrm{Com}(A))$ lorsque
    $A\in\GL_n(\R)$.
  \item Montrer que si $X$ est un vecteur propre de $A$ associé à une
    valeur propre non nulle, alors $X$ est un vecteur propre de
    $\big(\mathrm{Com}(A)\big)^T$.
  \end{enumerate}
   %exo 559 RMS 134-1
\end{exercice}

\begin{exercice}[concours=Mines-Ponts]
  Soit $n\in\N^*$. Soit $D$ l'ensemble des matrices $M\in\M_n(\K)$
  telles que $m_{i,j}=0$ si $i$ et $j$ sont de parités différentes.
  \begin{enumerate}[\bfseries a)]
  \item Montrer que $D$ est une sous-algèbre de $\M_n(\K)$.
  \item Soit $M\in D\cap\GL_n(\K)$. Montrer que $\mathrm{Com}(M)\in
    D$.
  \item Traiter le cas où $M$ n'est pas inversible.
  \end{enumerate}
   %exo 560 RMS 134-1
\end{exercise}
\begin{solution}
  \begin{enumerate}[\bfseries a)]
  \item Si $A,B\in, D$ et si $i,j$ de parité différente, $a_{i,k}$ ou
    $b_{k,j}$ est forcément nul donc $\dsum_{k=1}^na_{i,k}b_{k,j}=0$,
    ce qui prouve $AB\in D$.
  \end{enumerate}
\end{solution}
\end{enonce}
